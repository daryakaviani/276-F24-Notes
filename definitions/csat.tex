\begin{definition}[Circuit Satisfiability]
    We define the circuit satisfiability relation $\mathsf{CSAT}$
    over structure, instance, witness tuples
    as follows.
    \begin{equation*}
      \mathsf{CSAT}
      = 
      \left\{
      \begin{array}{l}
          % Statment, Witness
          (C, (x, y), w)
      \end{array}
      \middle\vert
      \begin{array}{l}
          C(x, w) = y
      \end{array}
      \right\}.
    \end{equation*}  
  \end{definition}
